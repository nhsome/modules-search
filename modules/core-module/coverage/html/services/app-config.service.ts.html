import { Inject, Injectable } from '@angular/core';
import { Http } from '@angular/http';
import { Observable } from 'rxjs/Rx';
import { IConfigService } from '../types/core.d.ts';
 
// Based on https://gist.github.com/fernandohu/122e88c3bcd210bbe41c608c36306db9
 
@Injectable()
export class AppConfig implements IConfigService {
 
    private config: any[] = null;
    private env: Object = null;
    private loaded: boolean = false;
 
    constructor(private http: Http) {
 
    }
 
    /**
     * Получить параметр из конфиг.файла
     * @key Ключ объекта в конфиг.файле
     * @subKey Дополнительный ключ для поиска в коллекции values обьекта с ключом key
     * @return Значение св-ва
     */
    public getConfig(key: any, subKey?: any) {
        this.checkLoaded();
        let conf = this.config.filter((c) => c.key === key)[0];
        if (!conf) {
          console.warn(`В файле config.${this.env['env']}.json не найден обьект с ключом ${key}`);
        }
 
        if (subKey) {
          let subObj = conf.values.filter((c) => c.key === subKey)[0];
          if (!subObj) {
            console.warn(`В объекте ${key} из config.${this.env['env']}.json не найден вложенный обьект с ключом ${subKey}`);
          }
          return subObj.value;
        }
 
        return conf.value;
      }
 
    /**
     * Чтение свойства из env файла
     * @key Ключ объекта в env файле
     * @return Значение св-ва
     */
    public getEnv(key: any) {
      this.checkLoaded();
      let envVar = this.env[key];
      if (!envVar) {
        console.warn(`В файле env.json не найден обьект с ключом ${key}`);
      }
      return envVar;
    }
 
    /**
     * Инициализация настроек приложения. Должен вызываться в APP_INITIALIZER
     * Читает env/env.json и env/config.[env].json
     */
    public load() {
        return new Promise((resolve, reject) => {
            this.http.get('env/env.json').map( (res) => res.json() ).catch((error: any): any => {
                console.log('Configuration file "env.json" could not be read');
                resolve(true);
                return Observable.throw(error.json().error || 'Server error');
            }).subscribe( (envResponse: any) => {
                this.env = envResponse;
                let request: any = null;
                request = this.http.get('env/config.' + envResponse.env + '.json');
                if (request) {
                    request
                        .map( (res) => res.json() )
                        .catch((error: any) => {
                            console.error('Error reading ' + envResponse.env + ' configuration file');
                            resolve(error);
                            return Observable.throw(error.json().error || 'Server error');
                        })
                        .subscribe((responseData) => {
                            this.config = responseData;
                            this.loaded = true;
                            resolve(true);
                        });
                } else {
                    console.error('Env config file "env.json" is not valid');
                    this.loaded = true;
                    resolve(true);
                }
            });
 
        });
    }
 
    private checkLoaded() {
      if (!this.loaded) {
        console.error(`You can't use AppConfig without call 'load()' method before.`);
      }
    }
}
